<script>
  // === CONFIG ===
  // Using an official YouTube upload for "Love in Portofino" (Andrea Bocelli / Universal)
  // Video ID used below (official uploads available on YouTube).
  // If you want a different upload, replace the videoId value.
  const VIDEO_ID = 'S_Bsk8FCfrs'; // (Andrea Bocelli / Universal) ‚Äî replace if needed

  // === YT PLAYER SETUP ===
  let player;
  function onYouTubeIframeAPIReady(){
    player = new YT.Player('player', {
      height: '1', width: '1',
      videoId: VIDEO_ID,
      playerVars: {
        autoplay: 0,
        controls: 1,
        modestbranding: 1,
        rel: 0,
        playsinline: 1
      },
      events:{
        'onReady': onPlayerReady
      }
    });
  }
  function onPlayerReady(ev){
    // nothing auto ‚Äî we'll call playVideo() after user taps
  }

  // === SEQUENCE ANIMATION ===
  const lines = [];
  for(let i=0;i<9;i++) lines.push(document.getElementById('l'+i));
  const overlay = document.getElementById('overlay');
  const startBtn = document.getElementById('startBtn');

  function showSequence(){
    const delays = [0,700,1400,2200,3000,3900,4700,5500,6300];
    for(let i=0;i<lines.length;i++){
      setTimeout(()=> {
        if(lines[i]) lines[i].classList.add('show');
        // trigger confetti on the "New update installed" line (index 4)
        if(i===4) launchConfetti(60);
      }, delays[i]);
    }
  }

  startBtn.addEventListener('click', () => {
    // hide overlay
    overlay.style.transition = 'opacity 350ms ease'; overlay.style.opacity = 0;
    setTimeout(()=> overlay.style.display = 'none', 400);

    // play youtube audio (user has interacted)
    try{
      if(player && player.playVideo) player.playVideo();
    }catch(e){
      console.warn('YT play failed, user may need to press play on the player.');
    }
    // begin text sequence
    setTimeout(showSequence, 220);
  });

  // === SIMPLE CONFETTI ===
  const confettiCanvas = document.getElementById('confetti');
  const ctx = confettiCanvas.getContext && confettiCanvas.getContext('2d');
  let confettiPieces = [];
  function resizeCanvas(){
    confettiCanvas.width = confettiCanvas.clientWidth || window.innerWidth;
    confettiCanvas.height = confettiCanvas.clientHeight || window.innerHeight;
  }
  window.addEventListener('resize', resizeCanvas);
  resizeCanvas();

  function launchConfetti(n){
    if(!ctx) return;
    for(let i=0;i<n;i++){
      confettiPieces.push({
        x: Math.random()*confettiCanvas.width,
        y: -20 - Math.random()*100,
        r: 6 + Math.random()*8,
        d: Math.random()*50+40,
        vx: (Math.random()-0.5)*2,
        vy: 1 + Math.random()*3,
        color: ['#f7c6b6','#f2e1c9','#d9bfa8','#ffd7a8'][Math.floor(Math.random()*4)],
        rot: Math.random()*360,
        vr: (Math.random()-0.5)*10
      });
    }
    if(!confettiLoopRunning) runConfettiLoop();
  }

  let confettiLoopRunning = false;
  function runConfettiLoop(){
    confettiLoopRunning = true;
    function frame(){
      ctx.clearRect(0,0,confettiCanvas.width,confettiCanvas.height);
      for(let i=0;i<confettiPieces.length;i++){
        const p = confettiPieces[i];
        p.x += p.vx;
        p.y += p.vy;
        p.rot += p.vr;
        ctx.save();
        ctx.translate(p.x,p.y);
        ctx.rotate(p.rot * Math.PI/180);
        ctx.fillStyle = p.color;
        ctx.fillRect(-p.r/2, -p.r/2, p.r, p.r*1.6);
        ctx.restore();

        if(p.y > confettiCanvas.height + 50){
          confettiPieces.splice(i,1); i--;
        }
      }
      if(confettiPieces.length>0) requestAnimationFrame(frame);
      else confettiLoopRunning = false;
    }
    requestAnimationFrame(frame);
  }

  // accessibility: allow pressing Enter on the start button
  startBtn.addEventListener('keydown', e=>{
    if(e.key==='Enter') startBtn.click();
  });

  // ensure canvas sizes right
  setTimeout(resizeCanvas, 200);
</script>
